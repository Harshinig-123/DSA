Q94
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<Integer> inorderTraversal(TreeNode root) {
        List l = new ArrayList<>();
        inorder(root,l);
        return l;
    }
    void inorder(TreeNode node, List<Integer> l)
    {
        if(node==null)
            return;
        {
            inorder(node.left,l);
            l.add(node.val);
            inorder(node.right,l);
        }
    }
}

class Solution {
    public List<Integer> postorderTraversal(TreeNode root) {
        List l = new ArrayList<>();
        postorder(root,l);
        return l;
    }
    void postorder(TreeNode node, List<Integer> l)
    {
        if(node==null)
            return;
        postorder(node.left,l);
        postorder(node.right,l);
        l.add(node.val);
    }
}

class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {
        List l = new ArrayList<>();
        preorder(root,l);
        return l;
    }
    void preorder(TreeNode node, List<Integer> l)
    {
        if(node == null)
            return;
        l.add(node.val);
        preorder(node.left,l);
        preorder(node.right,l);
    }
}

class Solution {
    public boolean isMonotonic(int[] nums) {
        return isIncreasing(nums)||isDecreasing(nums);
    }
    boolean isIncreasing(int[] nums)
    {   
        for(int i = 0; i<nums.length-1; i++)
        {
            if(nums[i]>nums[i+1])
                return false;
        }
        return true;
    }

    boolean isDecreasing(int[] nums)
    {
        for(int i = 0; i<nums.length-1;i++)
        {
            if(nums[i]<nums[i+1])
                return false;
        }
        return true;
    }
}
